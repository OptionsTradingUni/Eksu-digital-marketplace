// emailService.js

// 1. Import the Resend library
import { Resend } from 'resend';

// 2. Initialize Resend using the API Key from the environment
// The Resend constructor automatically looks for the RESEND_API_KEY environment variable.
const resend = new Resend();

/**
 * Sends a notification email using the Resend service.
 * It strictly uses the verified domain defined in RESEND_FROM_EMAIL.
 *
 * @param {string} to - The recipient's email address.
 * @param {string} subject - The subject line of the email.
 * @param {string} bodyHtml - The HTML body content of the email.
 * @returns {Promise<object>} - An object indicating success or failure.
 */
export async function sendNotificationEmail(to, subject, bodyHtml) {
    // CRITICAL: Get the verified sender address from the environment
    const fromEmail = process.env.RESEND_FROM_EMAIL;

    // Safety check to ensure the environment variable is set
    if (!fromEmail || !fromEmail.endsWith('@eksuplug.com.ng')) {
        console.error("üö´ RESEND_FROM_EMAIL is missing or invalid. Must use a verified domain!");
        return { success: false, error: "Missing or unverified sender configuration" };
    }

    try {
        console.log(`Attempting to send email from ${fromEmail} to ${to} via Resend...`);

        const { data, error } = await resend.emails.send({
            from: fromEmail, // <-- THIS IS THE FIXED SENDER ADDRESS
            to: [to],
            subject: subject,
            html: bodyHtml,
        });

        if (error) {
            console.error("‚ùå Resend send failed:", error);
            // The Resend error object will be returned to the calling function
            return { success: false, error: error };
        }

        console.log("‚úÖ Resend email sent successfully. ID:", data.id);
        return { success: true, data: data };

    } catch (e) {
        console.error("üö® An unexpected error occurred during Resend call:", e);
        return { success: false, error: e };
    }
}
