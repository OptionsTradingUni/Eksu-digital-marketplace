function verifySignature(payload: any, signature: string, secret: string): boolean {
  const crypto = require('crypto');
  const expected = crypto.createHmac('sha256', secret).update(JSON.stringify(payload)).digest('hex');
  return expected === signature;
}

  try:
    # Balance check

    if wallet.data[0]['balance'] < data['amount']:

    # Deduct
    supabase.table('wallets').update({'balance': wallet.data[0]['balance'] - data['amount']}).eq('user_id', data['user_id']).execute()

    # Payout
    payout = squad.payout.create(
      amount=data['amount'] * 100,
      bank_code=data['bank_code'],
      account_number=data['account_number'],
      account_name=data['account_name']
    )
    return jsonify(payout)
  except Exception as e:
    return jsonify({'error': str(e)}), 500

@app.route('/webhook/squad', methods=['POST'])
def squad_webhook():
  payload = request.json
  signature = request.headers.get('Squad-Signature')

  # Verify HMAC
  expected = hmac.new(
    os.getenv('SQUAD_WEBHOOK_SECRET').encode(),
    str(payload).encode(),
    hashlib.sha256
  ).hexdigest()

  if not hmac.compare_digest(signature, expected):
    return jsonify({'error': 'Invalid signature'}), 401

  if payload['event'] == 'charge.success':
    txn = payload['data']
    user_id = txn['customer']  # Map to your users

    # Credit wallet
    wallet = supabase.table('wallets').select('balance').eq('user_id', user_id).execute()
    new_balance = wallet.data[0]['balance'] + (txn['amount'] / 100)
    supabase.table('wallets').update({'balance': new_balance}).eq('user_id', user_id).execute()

    # Log
    supabase.table('transactions').insert({
      'reference': txn['reference'],
      'amount': txn['amount'] / 100,
      'type': 'deposit',
      'status': 'completed'
    }).execute()
